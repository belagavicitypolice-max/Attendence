Step 1: Create Google Apps Script

1. Open your Google Sheet

2. Click Extensions Apps Script

3. Delete any existing code and paste the following:

// Google Apps Script for Police Attendance System
const SHEET_ID = '1MA0MWmlTuWmDzsOsw7v0owf9hNlfCdfjlFoT4ZzbOJI';

function doPost(e) {
  return handleRequest(e);
}

function doGet(e) {
  return handleRequest(e);
}

function handleRequest(e) {
  try {
    const action = e.parameter.action;
    const sheet = SpreadsheetApp.openById(SHEET_ID);
    
    switch(action) {
      case 'submitAttendance':
        return submitAttendance(sheet, e.parameter);
      case 'addPersonnel':
        return addPersonnel(sheet, e.parameter);
      case 'getAttendance':
        return getAttendance(sheet, e.parameter);
      case 'getPersonnel':
        return getPersonnel(sheet);
      case 'checkDuplicate':
        return checkDuplicate(sheet, e.parameter);
      case 'getStats':
        return getStats(sheet);
      case 'exportData':
        return exportData(sheet, e.parameter);
      default:
        return createResponse('error', 'Invalid action');
    }
  } catch (error) {
    return createResponse('error', error.toString());
  }
}

function submitAttendance(sheet, data) {
  try {
    let attendanceSheet = sheet.getSheetByName('Attendance');
    if (!attendanceSheet) {
      attendanceSheet = sheet.insertSheet('Attendance');
      attendanceSheet.getRange(1, 1, 1, 11).setValues([[
        'Timestamp', 'Police ID', 'Name', 'Rank', 'Police Station', 
        'Sector', 'Shift', 'Latitude', 'Longitude', 'Address', 'Accuracy'
      ]]);
    }
    
    // Check for duplicate within last hour
    const now = new Date();
    const oneHourAgo = new Date(now.getTime() - 60 * 60 * 1000);
    const dataRange = attendanceSheet.getDataRange().getValues();
    
    for (let i = 1; i < dataRange.length; i++) {
      const row = dataRange[i];
      const rowTime = new Date(row[0]);
      if (row[1] === data.policeId && rowTime > oneHourAgo) {
        return createResponse('error', 'Attendance already marked within last hour');
      }
    }
    
    const timestamp = new Date();
    const rowData = [
      timestamp,
      data.policeId,
      data.fullName,
      data.rank,
      data.policeStation,
      data.sector,
      data.shift,
      data.latitude,
      data.longitude,
      data.address,
      data.accuracy
    ];
    
    attendanceSheet.appendRow(rowData);
    
    // Update Personnel sheet if needed
    updatePersonnelSheet(sheet, data);
    
    return createResponse('success', 'Attendance recorded successfully', {
      timestamp: timestamp.toISOString(),
      id: data.policeId
    });
  } catch (error) {
    return createResponse('error', error.toString());
  }
}

function addPersonnel(sheet, data) {
  try {
    let personnelSheet = sheet.getSheetByName('Personnel');
    if (!personnelSheet) {
      personnelSheet = sheet.insertSheet('Personnel');
      personnelSheet.getRange(1, 1, 1, 8).setValues([[
        'Police ID', 'Name', 'Rank', 'Police Station', 
        'Sector', 'Shift', 'Email', 'Contact'
      ]]);
    }
    
    // Check if ID already exists
    const dataRange = personnelSheet.getDataRange().getValues();
    for (let i = 1; i < dataRange.length; i++) {
      if (dataRange[i][0] === data.policeId) {
        return createResponse('error', 'Police ID already exists');
      }
    }
    
    const rowData = [
      data.policeId,
      data.fullName,
      data.rank,
      data.policeStation,
      data.sector,
      data.shift,
      data.email || '',
      data.contact || ''
    ];
    
    personnelSheet.appendRow(rowData);
    
    return createResponse('success', 'Personnel added successfully');
  } catch (error) {
    return createResponse('error', error.toString());
  }
}

function updatePersonnelSheet(sheet, data) {
  try {
    let personnelSheet = sheet.getSheetByName('Personnel');
    if (!personnelSheet) return;
    
    const dataRange = personnelSheet.getDataRange().getValues();
    let found = false;
    
    for (let i = 1; i < dataRange.length; i++) {
      if (dataRange[i][0] === data.policeId) {
        found = true;
        // Update existing record
        personnelSheet.getRange(i + 1, 2, 1, 5).setValues([[
          data.fullName,
          data.rank,
          data.policeStation,
          data.sector,
          data.shift
        ]]);
        break;
      }
    }
    
    if (!found) {
      // Add new personnel
      const rowData = [
        data.policeId,
        data.fullName,
        data.rank,
        data.policeStation,
        data.sector,
        data.shift,
        '',
        ''
      ];
      personnelSheet.appendRow(rowData);
    }
  } catch (error) {
    console.error('Error updating personnel sheet:', error);
  }
}

function getAttendance(sheet, params) {
  try {
    const attendanceSheet = sheet.getSheetByName('Attendance');
    if (!attendanceSheet) {
      return createResponse('success', { attendance: [] });
    }
    
    let data = attendanceSheet.getDataRange().getValues();
    const headers = data[0];
    
    // Apply filters if provided
    if (params.dateFrom || params.dateTo || params.station || params.sector || params.shift || params.rank) {
      data = data.filter((row, index) => {
        if (index === 0) return false; // Skip header
        
        if (params.dateFrom) {
          const rowDate = new Date(row[0]);
          const filterDate = new Date(params.dateFrom);
          if (rowDate < filterDate) return false;
        }
        
        if (params.dateTo) {
          const rowDate = new Date(row[0]);
          const filterDate = new Date(params.dateTo);
          filterDate.setHours(23, 59, 59, 999);
          if (rowDate > filterDate) return false;
        }
        
        if (params.station && row[4] !== params.station) return false;
        if (params.sector && row[5] !== params.sector) return false;
        if (params.shift && row[6] !== params.shift) return false;
        if (params.rank && row[3] !== params.rank) return false;
        
        return true;
      });
    } else {
      data = data.slice(1); // Remove header
    }
    
    // Format data
    const attendance = data.map(row => {
      const record = {};
      headers.forEach((header, index) => {
        record[header] = row[index];
      });
      return record;
    });
    
    // Get last 100 records if no filters
    const result = attendance.slice(-100).reverse();
    
    return createResponse('success', { attendance: result });
  } catch (error) {
    return createResponse('error', error.toString());
  }
}

function getPersonnel(sheet) {
  try {
    const personnelSheet = sheet.getSheetByName('Personnel');
    if (!personnelSheet) {
      return createResponse('success', { personnel: [] });
    }
    
    const data = personnelSheet.getDataRange().getValues();
    const personnel = data.slice(1).map(row => ({
      id: row[0],
      name: row[1],
      rank: row[2],
      station: row[3],
      sector: row[4],
      shift: row[5],
      email: row[6],
      contact: row[7]
    }));
    
    return createResponse('success', { personnel: personnel });
  } catch (error) {
    return createResponse('error', error.toString());
  }
}

function checkDuplicate(sheet, data) {
  try {
    const attendanceSheet = sheet.getSheetByName('Attendance');
    if (!attendanceSheet) {
      return createResponse('success', { isDuplicate: false });
    }
    
    const now = new Date();
    const oneHourAgo = new Date(now.getTime() - 60 * 60 * 1000);
    const dataRange = attendanceSheet.getDataRange().getValues();
    
    for (let i = 1; i < dataRange.length; i++) {
      const row = dataRange[i];
      const rowTime = new Date(row[0]);
      if (row[1] === data.policeId && rowTime > oneHourAgo) {
        return createResponse('success', { 
          isDuplicate: true,
          lastTime: rowTime.toISOString()
        });
      }
    }
    
    return createResponse('success', { isDuplicate: false });
  } catch (error) {
    return createResponse('error', error.toString());
  }
}

function getStats(sheet) {
  try {
    const attendanceSheet = sheet.getSheetByName('Attendance');
    const personnelSheet = sheet.getSheetByName('Personnel');
    
    let totalAttendance = 0;
    let totalPersonnel = 0;
    let presentToday = 0;
    let activeStations = new Set();
    const today = new Date().toDateString();
    
    if (attendanceSheet) {
      const attendanceData = attendanceSheet.getDataRange().getValues();
      totalAttendance = attendanceData.length - 1;
      
      for (let i = 1; i < attendanceData.length; i++) {
        const rowDate = new Date(attendanceData[i][0]).toDateString();
        if (rowDate === today) {
          presentToday++;
        }
        activeStations.add(attendanceData[i][4]);
      }
    }
    
    if (personnelSheet) {
      const personnelData = personnelSheet.getDataRange().getValues();
      totalPersonnel = personnelData.length - 1;
    }
    
    return createResponse('success', {
      totalAttendance: totalAttendance,
      totalPersonnel: totalPersonnel,
      presentToday: presentToday,
      activeStations: activeStations.size
    });
  } catch (error) {
    return createResponse('error', error.toString());
  }
}

function exportData(sheet, params) {
  try {
    const attendanceSheet = sheet.getSheetByName('Attendance');
    if (!attendanceSheet) {
      return createResponse('error', 'No attendance data found');
    }
    
    let data = attendanceSheet.getDataRange().getValues();
    const headers = data[0];
    
    // Apply filters
    if (params.dateFrom || params.dateTo || params.station || params.sector || params.shift || params.rank) {
      data = data.filter((row, index) => {
        if (index === 0) return true; // Keep header
        
        if (params.dateFrom) {
          const rowDate = new Date(row[0]);
          const filterDate = new Date(params.dateFrom);
          if (rowDate < filterDate) return false;
        }
        
        if (params.dateTo) {
          const rowDate = new Date(row[0]);
          const filterDate = new Date(params.dateTo);
          filterDate.setHours(23, 59, 59, 999);
          if (rowDate > filterDate) return false;
        }
        
        if (params.station && row[4] !== params.station) return false;
        if (params.sector && row[5] !== params.sector) return false;
        if (params.shift && row[6] !== params.shift) return false;
        if (params.rank && row[3] !== params.rank) return false;
        
        return true;
      });
    }
    
    // Convert to CSV
    const csv = data.map(row => 
      row.map(cell => {
        if (cell instanceof Date) {
          return cell.toLocaleString();
        }
        return `"${String(cell).replace(/"/g, '""')}"`;
      }).join(',')
    ).join('\n');
    
    return createResponse('success', { csv: csv });
  } catch (error) {
    return createResponse('error', error.toString());
  }
}

function createResponse(status, message, data = null) {
  const response = {
    status: status,
    message: message
  };
  
  if (data) {
    response.data = data;
  }
  
  return ContentService
    .createTextOutput(JSON.stringify(response))
    .setMimeType(ContentService.MimeType.JSON);
}
